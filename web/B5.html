Truy vấn dữ liệu

Sử dụng mysql để ghép dữ liệu từ 2 máy client 

CREATE TABLE marknice.client_1
(
	Title MEDIUMTEXT,
	Author MEDIUMTEXT,
	Genre MEDIUMTEXT,
	SubGenre MEDIUMTEXT,
	Height INT,
	Publisher MEDIUMTEXT
);
										

CREATE TABLE marknice.client_2
(
	Title MEDIUMTEXT,
	Author MEDIUMTEXT,
	Genre MEDIUMTEXT,
	SubGenre MEDIUMTEXT,
	Height INT,
	Publisher MEDIUMTEXT
);
Ghép dữ liệu 

CREATE TABLE 	marknice.host_data
(
	Title MEDIUMTEXT,
	Author MEDIUMTEXT,
	Genre MEDIUMTEXT,
	SubGenre MEDIUMTEXT,
	Height INT,
	Publisher MEDIUMTEXT
);
INSERT INTO marknice.host_data
(
	SELECT * FROM marknice.client_1
	UNION
	SELECT * FROM marknice.client_2
);
Kết quả Truy vấn

Truy vấn về thông tin tác giả quyển sách chưa rỏ Publisher

FROM marknice.host_data
